129/// Foundry configuration
130///
131/// # Defaults
132///
133/// All configuration values have a default, documented in the [fields](#fields)
134/// section below. [`Config::default()`] returns the default values for
135/// the default profile while [`Config::with_root()`] returns the values based on the given
136/// directory. [`Config::load()`] starts with the default profile and merges various providers into
137/// the config, same for [`Config::load_with_root()`], but there the default values are determined
138/// by [`Config::with_root()`]
139///
140/// # Provider Details
141///
142/// `Config` is a Figment [`Provider`] with the following characteristics:
143///
144///   * **Profile**
145///
146///     The profile is set to the value of the `profile` field.
147///
148///   * **Metadata**
149///
150///     This provider is named `Foundry Config`. It does not specify a
151///     [`Source`](figment::Source) and uses default interpolation.
152///
153///   * **Data**
154///
155///     The data emitted by this provider are the keys and values corresponding
156///     to the fields and values of the structure. The dictionary is emitted to
157///     the "default" meta-profile.
158///
159/// Note that these behaviors differ from those of [`Config::figment()`].
160#[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]

645    fn from_figment(figment: Figment) -> Result<Self, ExtractConfigError> {
646        let mut config = figment.extract::<Self>().map_err(ExtractConfigError::new)?;
647        config.profile = figment.profile().clone();
648
649        // The `"profile"` profile contains all the profiles as keys.
650        let mut add_profile = |profile: &Profile| {
651            if !config.profiles.contains(profile) {
652                config.profiles.push(profile.clone());
653            }
654        };
655        let figment = figment.select(Self::PROFILE_SECTION);
656        if let Ok(data) = figment.data() {
657            if let Some(profiles) = data.get(&Profile::new(Self::PROFILE_SECTION)) {
658                for profile in profiles.keys() {
659                    add_profile(&Profile::new(profile));
660                }
661            }
662        }
663        add_profile(&Self::DEFAULT_PROFILE);
664        add_profile(&config.profile);
665
666        config.normalize_optimizer_settings();
667
668        Ok(config)
669    }

671    /// Returns the populated [Figment] using the requested [FigmentProviders] preset.
672    ///
673    /// This will merge various providers, such as env,toml,remappings into the figment if
674    /// requested.
675    pub fn to_figment(&self, providers: FigmentProviders) -> Figment {
676        // Note that `Figment::from` here is a method on `Figment` rather than the `From` impl below
677
678        if providers.is_none() {
679            return Figment::from(self);
680        }
681
682        let root = self.root.as_path();
683        let profile = Self::selected_profile();
684        let mut figment = Figment::default().merge(DappHardhatDirProvider(root));
685
686        // merge global foundry.toml file
687        if let Some(global_toml) = Self::foundry_dir_toml().filter(|p| p.exists()) {
688            figment = Self::merge_toml_provider(
689                figment,
690                TomlFileProvider::new(None, global_toml).cached(),
691                profile.clone(),
692            );
693        }
694        // merge local foundry.toml file
695        figment = Self::merge_toml_provider(
696            figment,
697            TomlFileProvider::new(Some("FOUNDRY_CONFIG"), root.join(Self::FILE_NAME)).cached(),
698            profile.clone(),
699        );
700
701        // merge environment variables
702        figment = figment
703            .merge(
704                Env::prefixed("DAPP_")
705                    .ignore(&["REMAPPINGS", "LIBRARIES", "FFI", "FS_PERMISSIONS"])
706                    .global(),
707            )
708            .merge(
709                Env::prefixed("DAPP_TEST_")
710                    .ignore(&["CACHE", "FUZZ_RUNS", "DEPTH", "FFI", "FS_PERMISSIONS"])
711                    .global(),
712            )
713            .merge(DappEnvCompatProvider)
714            .merge(EtherscanEnvProvider::default())
715            .merge(
716                Env::prefixed("FOUNDRY_")
717                    .ignore(&["PROFILE", "REMAPPINGS", "LIBRARIES", "FFI", "FS_PERMISSIONS"])
718                    .map(|key| {
719                        let key = key.as_str();
720                        if Self::STANDALONE_SECTIONS.iter().any(|section| {
721                            key.starts_with(&format!("{}_", section.to_ascii_uppercase()))
722                        }) {
723                            key.replacen('_', ".", 1).into()
724                        } else {
725                            key.into()
726                        }
727                    })
728                    .global(),
729            )
730            .select(profile.clone());
731
732        // only resolve remappings if all providers are requested
733        if providers.is_all() {
734            // we try to merge remappings after we've merged all other providers, this prevents
735            // redundant fs lookups to determine the default remappings that are eventually updated
736            // by other providers, like the toml file
737            let remappings = RemappingsProvider {
738                auto_detect_remappings: figment
739                    .extract_inner::<bool>("auto_detect_remappings")
740                    .unwrap_or(true),
741                lib_paths: figment
742                    .extract_inner::<Vec<PathBuf>>("libs")
743                    .map(Cow::Owned)
744                    .unwrap_or_else(|_| Cow::Borrowed(&self.libs)),
745                root,
746                remappings: figment.extract_inner::<Vec<Remapping>>("remappings"),
747            };
748            figment = figment.merge(remappings);
749        }
750
751        // normalize defaults
752        figment = self.normalize_defaults(figment);
753
754        Figment::from(self).merge(figment).select(profile)
755    }

767    /// Joins all relative paths with the given root so that paths that are defined as:
768    ///
769    /// ```toml
770    /// [profile.default]
771    /// src = "src"
772    /// out = "./out"
773    /// libs = ["lib", "/var/lib"]
774    /// ```
775    ///
776    /// Will be made canonic with the given root:
777    ///
778    /// ```toml
779    /// [profile.default]
780    /// src = "<root>/src"
781    /// out = "<root>/out"
782    /// libs = ["<root>/lib", "/var/lib"]
783    /// ```
784    #[must_use]
785    pub fn canonic_at(mut self, root: impl Into<PathBuf>) -> Self {
786        let root = canonic(root);
787
788        fn p(root: &Path, rem: &Path) -> PathBuf {
789            canonic(root.join(rem))
790        }
791
792        self.src = p(&root, &self.src);
793        self.test = p(&root, &self.test);
794        self.script = p(&root, &self.script);
795        self.out = p(&root, &self.out);
796        self.broadcast = p(&root, &self.broadcast);
797        self.cache_path = p(&root, &self.cache_path);
798        self.snapshots = p(&root, &self.snapshots);
799
800        if let Some(build_info_path) = self.build_info_path {
801            self.build_info_path = Some(p(&root, &build_info_path));
802        }
803
804        self.libs = self.libs.into_iter().map(|lib| p(&root, &lib)).collect();
805
806        self.remappings =
807            self.remappings.into_iter().map(|r| RelativeRemapping::new(r.into(), &root)).collect();
808
809        self.allow_paths = self.allow_paths.into_iter().map(|allow| p(&root, &allow)).collect();
810
811        self.include_paths = self.include_paths.into_iter().map(|allow| p(&root, &allow)).collect();
812
813        self.fs_permissions.join_all(&root);
814
815        if let Some(model_checker) = &mut self.model_checker {
816            model_checker.contracts = std::mem::take(&mut model_checker.contracts)
817                .into_iter()
818                .map(|(path, contracts)| {
819                    (format!("{}", p(&root, path.as_ref()).display()), contracts)
820                })
821                .collect();
822        }
823
824        self
825    }

859            (None, Some(runs)) => self.optimizer = Some(runs > 0),
860            _ => {}
861        }

955    /// Resolves globs and builds a mapping from individual source files to their restrictions
956    #[expect(clippy::disallowed_macros)]
957    fn restrictions(
958        &self,
959        paths: &ProjectPathsConfig,
960    ) -> Result<BTreeMap<PathBuf, RestrictionsWithVersion<MultiCompilerRestrictions>>, SolcError>
961    {
962        let mut map = BTreeMap::new();
963        if self.compilation_restrictions.is_empty() {
964            return Ok(BTreeMap::new());
965        }
966
967        let graph = Graph::<MultiCompilerParsedSource>::resolve(paths)?;
968        let (sources, _) = graph.into_sources();
969
970        for res in &self.compilation_restrictions {
971            for source in sources.keys().filter(|path| {
972                if res.paths.is_match(path) {
973                    true
974                } else if let Ok(path) = path.strip_prefix(&paths.root) {
975                    res.paths.is_match(path)
976                } else {
977                    false
978                }
979            }) {
980                let res: RestrictionsWithVersion<_> =
981                    res.clone().try_into().map_err(SolcError::msg)?;
982                if !map.contains_key(source) {
983                    map.insert(source.clone(), res);
984                } else {
985                    let value = map.remove(source.as_path()).unwrap();
986                    if let Some(merged) = value.clone().merge(res) {
987                        map.insert(source.clone(), merged);
988                    } else {
989                        // `sh_warn!` is a circular dependency, preventing us from using it here.
990                        eprintln!(
991                            "{}",
992                            yansi::Paint::yellow(&format!(
993                                "Failed to merge compilation restrictions for {}",
994                                source.display()
995                            ))
996                        );
997                        map.insert(source.clone(), value);
998                    }
999                }
1000            }
1001        }
1002
1003        Ok(map)
1004    }

1006    /// Creates a [`Project`] with the given `cached` and `no_artifacts` flags.
1007    ///
1008    /// Prefer using [`Self::project`] or [`Self::ephemeral_project`] instead.
1009    pub fn create_project(&self, cached: bool, no_artifacts: bool) -> Result<Project, SolcError> {
1010        let settings = self.compiler_settings()?;
1011        let paths = self.project_paths();
1012        let mut builder = Project::builder()
1013            .artifacts(self.configured_artifacts_handler())
1014            .additional_settings(self.additional_settings(&settings))
1015            .restrictions(self.restrictions(&paths)?)
1016            .settings(settings)
1017            .paths(paths)
1018            .ignore_error_codes(self.ignored_error_codes.iter().copied().map(Into::into))
1019            .ignore_paths(self.ignored_file_paths.clone())
1020            .set_compiler_severity_filter(if self.deny_warnings {
1021                Severity::Warning
1022            } else {
1023                Severity::Error
1024            })
1025            .set_offline(self.offline)
1026            .set_cached(cached)
1027            .set_build_info(!no_artifacts && self.build_info)
1028            .set_no_artifacts(no_artifacts);
1029
1030        if !self.skip.is_empty() {
1031            let filter = SkipBuildFilters::new(self.skip.clone(), self.root.clone());
1032            builder = builder.sparse_output(filter);
1033        }
1034
1035        let project = builder.build(self.compiler()?)?;
1036
1037        if self.force {
1038            self.cleanup(&project)?;
1039        }
1040
1041        Ok(project)
1042    }

1087                        }
1088                        Solc::blocking_install(version)?
1089                    }
1090                }
1091                SolcReq::Local(solc) => {
1092                    if !solc.is_file() {
1093                        return Err(SolcError::msg(format!(
1094                            "`solc` {} does not exist",
1095                            solc.display()
1096                        )));
1097                    }
1098                    Solc::new(solc)?
1099                }
1100            };
1101            return Ok(Some(solc));
1102        }

1104        Ok(None)
1105    }

1251    /// Resolves the given alias to a matching rpc url
1252    ///
1253    /// # Returns
1254    ///
1255    /// In order of resolution:
1256    ///
1257    /// - the matching, resolved url of `rpc_endpoints` if `maybe_alias` is an alias
1258    /// - a mesc resolved url if `maybe_alias` is a known alias in mesc
1259    /// - `None` otherwise
1260    ///
1261    /// # Note on mesc
1262    ///
1263    /// The endpoint is queried for in mesc under the `foundry` profile, allowing users to customize
1264    /// endpoints for Foundry specifically.
1265    ///
1266    /// # Example
1267    ///
1268    /// ```
1269    /// use foundry_config::Config;
1270    /// # fn t() {
1271    /// let config = Config::with_root("./");
1272    /// let rpc_url = config.get_rpc_url_with_alias("mainnet").unwrap().unwrap();
1273    /// # }
1274    /// ```
1275    pub fn get_rpc_url_with_alias(
1276        &self,
1277        maybe_alias: &str,
1278    ) -> Option<Result<Cow<'_, str>, UnresolvedEnvVarError>> {
1279        let mut endpoints = self.rpc_endpoints.clone().resolved();
1280        if let Some(endpoint) = endpoints.remove(maybe_alias) {
1281            return Some(endpoint.url().map(Cow::Owned));
1282        }
1283
1284        if let Ok(Some(endpoint)) = mesc::get_endpoint_by_query(maybe_alias, Some("foundry")) {
1285            return Some(Ok(Cow::Owned(endpoint.url)));
1286        }
1287
1288        None
1289    }

1353    /// Same as [`Self::get_etherscan_config()`] but optionally updates the config with the given
1354    /// `chain`, and `etherscan_api_key`
1355    ///
1356    /// If not matching alias was found, then this will try to find the first entry in the table
1357    /// with a matching chain id. If an etherscan_api_key is already set it will take precedence
1358    /// over the chain's entry in the table.
1359    pub fn get_etherscan_config_with_chain(
1360        &self,
1361        chain: Option<Chain>,
1362    ) -> Result<Option<ResolvedEtherscanConfig>, EtherscanConfigError> {
1363        let default_api_version = self.etherscan_api_version.unwrap_or_default();
1364
1365        if let Some(maybe_alias) = self.etherscan_api_key.as_ref().or(self.eth_rpc_url.as_ref()) {
1366            if self.etherscan.contains_key(maybe_alias) {
1367                return self
1368                    .etherscan
1369                    .clone()
1370                    .resolved(default_api_version)
1371                    .remove(maybe_alias)
1372                    .transpose();
1373            }
1374        }
1375
1376        // try to find by comparing chain IDs after resolving
1377        if let Some(res) = chain.or(self.chain).and_then(|chain| {
1378            self.etherscan.clone().resolved(default_api_version).find_chain(chain)
1379        }) {
1380            match (res, self.etherscan_api_key.as_ref()) {
1381                (Ok(mut config), Some(key)) => {
1382                    // we update the key, because if an etherscan_api_key is set, it should take
1383                    // precedence over the entry, since this is usually set via env var or CLI args.
1384                    config.key.clone_from(key);
1385                    return Ok(Some(config));
1386                }
1387                (Ok(config), None) => return Ok(Some(config)),
1388                (Err(err), None) => return Err(err),
1389                (Err(_), Some(_)) => {
1390                    // use the etherscan key as fallback
1391                }
1392            }
1393        }
1394
1395        // etherscan fallback via API key
1396        if let Some(key) = self.etherscan_api_key.as_ref() {
1397            return Ok(ResolvedEtherscanConfig::create(
1398                key,
1399                chain.or(self.chain).unwrap_or_default(),
1400                default_api_version,
1401            ));
1402        }
1403
1404        Ok(None)
1405    }

1500    /// - all libraries

1502    /// - evm version
1503    pub fn solc_settings(&self) -> Result<SolcSettings, SolcError> {
1504        // By default if no targets are specifically selected the model checker uses all targets.
1505        // This might be too much here, so only enable assertion checks.

1512        }

1515            libraries: self.libraries_with_remappings()?,
1516            optimizer: self.optimizer(),
1517            evm_version: Some(self.evm_version),
1518            metadata: Some(SettingsMetadata {

1521                cbor_metadata: Some(self.cbor_metadata),
1522            }),
1523            debug: self.revert_strings.map(|revert_strings| DebuggingSettings {
1524                revert_strings: Some(revert_strings),
1525                // Not used.
1526                debug_info: Vec::new(),
1527            }),
1528            model_checker,
1529            via_ir: Some(self.via_ir),
1530            // Not used.
1531            stop_after: None,
1532            // Set in project paths.
1533            remappings: Vec::new(),
1534            // Set with `with_extra_output` below.
1535            output_selection: Default::default(),

1537        .with_extra_output(self.configured_artifacts_handler().output_selection());

1539        // We're keeping AST in `--build-info` for backwards compatibility with HardHat.
1540        if self.ast || self.build_info {
1541            settings = settings.with_ast();
1542        }
1543
1544        let cli_settings =
1545            CliSettings { extra_args: self.extra_args.clone(), ..Default::default() };

1743    /// Serialize the config type as a String of TOML.
1744    ///
1745    /// This serializes to a table with the name of the profile
1746    ///
1747    /// ```toml
1748    /// [profile.default]
1749    /// src = "src"
1750    /// out = "out"
1751    /// libs = ["lib"]
1752    /// # ...
1753    /// ```
1754    pub fn to_string_pretty(&self) -> Result<String, toml::ser::Error> {
1755        // serializing to value first to prevent `ValueAfterTable` errors
1756        let mut value = toml::Value::try_from(self)?;
1757        // Config map always gets serialized as a table
1758        let value_table = value.as_table_mut().unwrap();
1759        // remove standalone sections from inner table
1760        let standalone_sections = Self::STANDALONE_SECTIONS
1761            .iter()
1762            .filter_map(|section| {
1763                let section = section.to_string();
1764                value_table.remove(&section).map(|value| (section, value))
1765            })
1766            .collect::<Vec<_>>();
1767        // wrap inner table in [profile.<profile>]
1768        let mut wrapping_table = [(
1769            Self::PROFILE_SECTION.into(),
1770            toml::Value::Table([(self.profile.to_string(), value)].into_iter().collect()),
1771        )]
1772        .into_iter()
1773        .collect::<toml::map::Map<_, _>>();
1774        // insert standalone sections
1775        for (section, value) in standalone_sections {
1776            wrapping_table.insert(section, value);
1777        }
1778        // stringify
1779        toml::to_string_pretty(&toml::Value::Table(wrapping_table))
1780    }

2048    fn merge_toml_provider(
2049        mut figment: Figment,
2050        toml_provider: impl Provider,
2051        profile: Profile,
2052    ) -> Figment {
2053        figment = figment.select(profile.clone());
2054
2055        // add warnings
2056        figment = {
2057            let warnings = WarningsProvider::for_figment(&toml_provider, &figment);
2058            figment.merge(warnings)
2059        };
2060
2061        // use [profile.<profile>] as [<profile>]
2062        let mut profiles = vec![Self::DEFAULT_PROFILE];
2063        if profile != Self::DEFAULT_PROFILE {
2064            profiles.push(profile.clone());
2065        }
2066        let provider = toml_provider.strict_select(profiles);
2067
2068        // apply any key fixes
2069        let provider = &BackwardsCompatTomlProvider(ForcedSnakeCaseData(provider));
2070
2071        // merge the default profile as a base
2072        if profile != Self::DEFAULT_PROFILE {
2073            figment = figment.merge(provider.rename(Self::DEFAULT_PROFILE, profile.clone()));
2074        }
2075        // merge special keys into config
2076        for standalone_key in Self::STANDALONE_SECTIONS {
2077            if let Some((_, fallback)) =
2078                STANDALONE_FALLBACK_SECTIONS.iter().find(|(key, _)| standalone_key == key)
2079            {
2080                figment = figment.merge(
2081                    provider
2082                        .fallback(standalone_key, fallback)
2083                        .wrap(profile.clone(), standalone_key),
2084                );
2085            } else {
2086                figment = figment.merge(provider.wrap(profile.clone(), standalone_key));
2087            }
2088        }
2089        // merge the profile
2090        figment = figment.merge(provider);
2091        figment
2092    }

2431/// `i64`
2432#[derive(Clone, Copy, Debug, Default, PartialEq, Eq, Deserialize)]
2433pub struct GasLimit(#[serde(deserialize_with = "crate::deserialize_u64_or_max")] pub u64);
2434
2435impl From<u64> for GasLimit {
2436    fn from(gas: u64) -> Self {
2437        Self(gas)
2438    }
2439}
2440
2441impl From<GasLimit> for u64 {
2442    fn from(gas: GasLimit) -> Self {
2443        gas.0
2444    }
2445}
2446
2447impl Serialize for GasLimit {
2448    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
2449    where
2450        S: Serializer,
2451    {
2452        if self.0 == u64::MAX {
2453            serializer.serialize_str("max")
2454        } else if self.0 > i64::MAX as u64 {
2455            serializer.serialize_str(&self.0.to_string())
2456        } else {
2457            serializer.serialize_u64(self.0)
2458        }
2459    }
2460}

2462/// Variants for selecting the [`Solc`] instance
2463#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
2464#[serde(untagged)]
2465pub enum SolcReq {
2466    /// Requires a specific solc version, that's either already installed (via `svm`) or will be
2467    /// auto installed (via `svm`)
2468    Version(Version),
2469    /// Path to an existing local solc installation
2470    Local(PathBuf),
2471}
2472
2473impl SolcReq {
2474    /// Tries to get the solc version from the `SolcReq`
2475    ///
2476    /// If the `SolcReq` is a `Version` it will return the version, if it's a path to a binary it
2477    /// will try to get the version from the binary.
2478    fn try_version(&self) -> Result<Version, SolcError> {
2479        match self {
2480            Self::Version(version) => Ok(version.clone()),
2481            Self::Local(path) => Solc::new(path).map(|solc| solc.version),
2482        }
2483    }
2484}
2485
2486impl<T: AsRef<str>> From<T> for SolcReq {
2487    fn from(s: T) -> Self {
2488        let s = s.as_ref();
2489        if let Ok(v) = Version::from_str(s) {
2490            Self::Version(v)
2491        } else {
2492            Self::Local(s.into())
2493        }
2494    }
2495}
