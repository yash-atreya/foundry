44/// Useful extensions to [`std::path::Path`].
45pub trait FoundryPathExt {
46    /// Returns true if the [`Path`] ends with `.t.sol`
47    fn is_sol_test(&self) -> bool;
48
49    /// Returns true if the  [`Path`] has a `sol` extension
50    fn is_sol(&self) -> bool;
51
52    /// Returns true if the  [`Path`] has a `yul` extension
53    fn is_yul(&self) -> bool;
54}
55
56impl<T: AsRef<Path>> FoundryPathExt for T {
57    fn is_sol_test(&self) -> bool {
58        self.as_ref()
59            .file_name()
60            .and_then(|s| s.to_str())
61            .map(|s| s.ends_with(".t.sol"))
62            .unwrap_or_default()
63    }
64
65    fn is_sol(&self) -> bool {
66        self.as_ref().extension() == Some(std::ffi::OsStr::new("sol"))
67    }
68
69    fn is_yul(&self) -> bool {
70        self.as_ref().extension() == Some(std::ffi::OsStr::new("yul"))
71    }
72}

95/// Returns a [ProviderBuilder] instantiated using [Config] values.
96///
97/// Defaults to `http://localhost:8545` and `Mainnet`.
98pub fn get_provider_builder(config: &Config) -> Result<ProviderBuilder> {
99    let url = config.get_rpc_url_or_localhost_http()?;
100    let mut builder = ProviderBuilder::new(url.as_ref());
101
102    builder = builder.accept_invalid_certs(config.eth_rpc_accept_invalid_certs);
103
104    if let Ok(chain) = config.chain.unwrap_or_default().try_into() {
105        builder = builder.chain(chain);
106    }
107
108    if let Some(jwt) = config.get_rpc_jwt_secret()? {
109        builder = builder.jwt(jwt.as_ref());
110    }
111
112    if let Some(rpc_timeout) = config.eth_rpc_timeout {
113        builder = builder.timeout(Duration::from_secs(rpc_timeout));
114    }
115
116    if let Some(rpc_headers) = config.eth_rpc_headers.clone() {
117        builder = builder.headers(rpc_headers);
118    }
119
120    Ok(builder)
121}

133/// Parses an ether value from a string.
134///
135/// The amount can be tagged with a unit, e.g. "1ether".
136///
137/// If the string represents an untagged amount (e.g. "100") then
138/// it is interpreted as wei.
139pub fn parse_ether_value(value: &str) -> Result<U256> {
140    Ok(if value.starts_with("0x") {
141        U256::from_str_radix(value, 16)?
142    } else {
143        alloy_dyn_abi::DynSolType::coerce_str(&alloy_dyn_abi::DynSolType::Uint(256), value)?
144            .as_uint()
145            .wrap_err("Could not parse ether value from string")?
146            .0
147    })
148}

155/// Parses a `Duration` from a &str
156pub fn parse_delay(delay: &str) -> Result<Duration> {
157    let delay = if delay.ends_with("ms") {
158        let d: u64 = delay.trim_end_matches("ms").parse()?;
159        Duration::from_millis(d)
160    } else {
161        let d: f64 = delay.parse()?;
162        let delay = (d * 1000.0).round();
163        if delay.is_infinite() || delay.is_nan() || delay.is_sign_negative() {
164            eyre::bail!("delay must be finite and non-negative");
165        }
166
167        Duration::from_millis(delay as u64)
168    };
169    Ok(delay)
170}

183/// Loads a dotenv file, from the cwd and the project root, ignoring potential failure.
184///
185/// We could use `warn!` here, but that would imply that the dotenv file can't configure
186/// the logging behavior of Foundry.
187///
188/// Similarly, we could just use `eprintln!`, but colors are off limits otherwise dotenv is implied
189/// to not be able to configure the colors. It would also mess up the JSON output.
190pub fn load_dotenv() {
191    let load = |p: &Path| {
192        dotenvy::from_path(p.join(".env")).ok();
193    };
194
195    // we only want the .env file of the cwd and project root
196    // `find_project_root` calls `current_dir` internally so both paths are either both `Ok` or
197    // both `Err`
198    if let (Ok(cwd), Ok(prj_root)) = (std::env::current_dir(), find_project_root(None)) {
199        load(&prj_root);
200        if cwd != prj_root {
201            // prj root and cwd can be identical
202            load(&cwd);
203        }
204    };
205}

230/// Useful extensions to [`std::process::Command`].
231pub trait CommandUtils {
232    /// Returns the command's output if execution is successful, otherwise, throws an error.
233    fn exec(&mut self) -> Result<Output>;
234
235    /// Returns the command's stdout if execution is successful, otherwise, throws an error.
236    fn get_stdout_lossy(&mut self) -> Result<String>;
237}
238
239impl CommandUtils for Command {
240    #[track_caller]
241    fn exec(&mut self) -> Result<Output> {
242        trace!(command=?self, "executing");
243
244        let output = self.output()?;
245
246        trace!(code=?output.status.code(), ?output);
247
248        if output.status.success() {
249            Ok(output)
250        } else {
251            let stdout = String::from_utf8_lossy(&output.stdout);
252            let stdout = stdout.trim();
253            let stderr = String::from_utf8_lossy(&output.stderr);
254            let stderr = stderr.trim();
255            let msg = if stdout.is_empty() {
256                stderr.to_string()
257            } else if stderr.is_empty() {
258                stdout.to_string()
259            } else {
260                format!("stdout:\n{stdout}\n\nstderr:\n{stderr}")
261            };
262
263            let mut name = self.get_program().to_string_lossy();
264            if let Some(arg) = self.get_args().next() {
265                let arg = arg.to_string_lossy();
266                if !arg.starts_with('-') {
267                    let name = name.to_mut();
268                    name.push(' ');
269                    name.push_str(&arg);
270                }
271            }
272
273            let mut err = match output.status.code() {
274                Some(code) => format!("{name} exited with code {code}"),
275                None => format!("{name} terminated by a signal"),
276            };
277            if !msg.is_empty() {
278                err.push(':');
279                err.push(if msg.lines().count() == 0 { ' ' } else { '\n' });
280                err.push_str(&msg);
281            }
282            Err(eyre::eyre!(err))
283        }
284    }
285
286    #[track_caller]
287    fn get_stdout_lossy(&mut self) -> Result<String> {
288        let output = self.exec()?;
289        let stdout = String::from_utf8_lossy(&output.stdout);
290        Ok(stdout.trim().into())
291    }
292}

435    pub fn commit(self, msg: &str) -> Result<()> {
436        let output = self
437            .cmd()
438            .args(["commit", "-m", msg])
439            .args(cfg!(any(test, debug_assertions)).then_some("--no-gpg-sign"))
440            .output()?;
441        if !output.status.success() {
442            let stdout = String::from_utf8_lossy(&output.stdout);
443            let stderr = String::from_utf8_lossy(&output.stderr);
444            // ignore "nothing to commit" error
445            let msg = "nothing to commit, working tree clean";
446            if !(stdout.contains(msg) || stderr.contains(msg)) {
447                return Err(eyre::eyre!(
448                    "failed to commit (code={:?}, stdout={:?}, stderr={:?})",
449                    output.status.code(),
450                    stdout.trim(),
451                    stderr.trim()
452                ));
453            }
454        }
455        Ok(())
456    }
