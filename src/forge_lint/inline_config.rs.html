5/// An inline config item
6#[derive(Clone, Debug)]
7pub enum InlineConfigItem {
8    /// Disables the next code (AST) item regardless of newlines
9    DisableNextItem(Vec<String>),
10    /// Disables formatting on the current line
11    DisableLine(Vec<String>),
12    /// Disables formatting between the next newline and the newline after
13    DisableNextLine(Vec<String>),
14    /// Disables formatting for any code that follows this and before the next "disable-end"
15    DisableStart(Vec<String>),
16    /// Disables formatting for any code that precedes this and after the previous "disable-start"
17    DisableEnd(Vec<String>),
18}
19
20impl InlineConfigItem {
21    /// Parse an inline config item from a string. Validates lint IDs against available lints.
22    pub fn parse(s: &str, available_lints: &[&str]) -> Result<Self, InvalidInlineConfigItem> {
23        let (disable, relevant) = s.split_once('(').unwrap_or((s, ""));
24        let lints = if relevant.is_empty() || relevant == "all)" {
25            vec!["all".to_string()]
26        } else {
27            match relevant.split_once(')') {
28                Some((lint, _)) => lint.split(",").map(|s| s.trim().to_string()).collect(),
29                None => return Err(InvalidInlineConfigItem::Syntax(s.into())),
30            }
31        };
32
33        // Validate lint IDs
34        let mut invalid_ids = Vec::new();
35        'ids: for id in &lints {
36            if id == "all" {
37                continue;
38            }
39            for lint in available_lints {
40                if *lint == id {
41                    continue 'ids;
42                }
43            }
44            invalid_ids.push(id.to_owned());
45        }
46
47        if !invalid_ids.is_empty() {
48            return Err(InvalidInlineConfigItem::LintIds(invalid_ids));
49        }
50
51        let res = match disable {
52            "disable-next-item" => Self::DisableNextItem(lints),
53            "disable-line" => Self::DisableLine(lints),
54            "disable-next-line" => Self::DisableNextLine(lints),
55            "disable-start" => Self::DisableStart(lints),
56            "disable-end" => Self::DisableEnd(lints),
57            s => return Err(InvalidInlineConfigItem::Syntax(s.into())),
58        };
59
60        Ok(res)
61    }
62}

64#[derive(Debug)]
65pub enum InvalidInlineConfigItem {
66    Syntax(String),
67    LintIds(Vec<String>),
68}
69
70impl fmt::Display for InvalidInlineConfigItem {
71    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
72        match self {
73            Self::Syntax(s) => write!(f, "invalid inline config item: {s}"),
74            Self::LintIds(ids) => {
75                write!(f, "unknown lint id: '{}'", ids.join("', '"))
76            }
77        }
78    }
79}

81/// A disabled formatting range. `loose` designates that the range includes any loc which
82/// may start in between start and end, whereas the strict version requires that
83/// `range.start >= loc.start <=> loc.end <= range.end`
84#[derive(Debug, Clone, Copy)]
85struct DisabledRange {
86    start: usize,
87    end: usize,
88    loose: bool,
89}
90
91impl DisabledRange {
92    fn includes(&self, range: std::ops::Range<usize>) -> bool {
93        range.start >= self.start && (if self.loose { range.start } else { range.end } <= self.end)
94    }
95}

97/// An inline config. Keeps track of ranges which should not be formatted.
98#[derive(Debug, Default)]
99pub struct InlineConfig {
100    disabled_ranges: HashMap<String, Vec<DisabledRange>>,
101}
102
103impl InlineConfig {
104    /// Build a new inline config with an iterator of inline config items and their locations in a
105    /// source file.
106    ///
107    /// # Panics
108    ///
109    /// Panics if `items` is not sorted in ascending order of [`Span`]s.
110    pub fn new<'ast>(
111        items: impl IntoIterator<Item = (Span, InlineConfigItem)>,
112        ast: &'ast SourceUnit<'ast>,
113        src: &str,
114    ) -> Self {
115        let mut disabled_ranges: HashMap<String, Vec<DisabledRange>> = HashMap::new();
116        let mut disabled_blocks: HashMap<String, (usize, usize)> = HashMap::new();
117
118        let mut prev_sp = Span::DUMMY;
119        for (sp, item) in items {
120            if cfg!(debug_assertions) {
121                assert!(sp >= prev_sp, "InlineConfig::new: unsorted items: {sp:?} < {prev_sp:?}");
122                prev_sp = sp;
123            }
124
125            match item {
126                InlineConfigItem::DisableNextItem(lints) => {
127                    let comment_end = sp.hi().to_usize();
128
129                    if let Some(next_item) = NextItemFinder::new(comment_end).find(ast) {
130                        for lint in lints {
131                            disabled_ranges.entry(lint).or_default().push(DisabledRange {
132                                start: next_item.lo().to_usize(),
133                                end: next_item.hi().to_usize(),
134                                loose: false,
135                            });
136                        }
137                    };
138                }
139                InlineConfigItem::DisableLine(lints) => {
140                    let mut prev_newline = src[..sp.lo().to_usize()]
141                        .char_indices()
142                        .rev()
143                        .skip_while(|(_, ch)| *ch != '\n');
144                    let start = prev_newline.next().map(|(idx, _)| idx).unwrap_or_default();
145
146                    let end_offset = sp.hi().to_usize();
147                    let mut next_newline =
148                        src[end_offset..].char_indices().skip_while(|(_, ch)| *ch != '\n');
149                    let end =
150                        end_offset + next_newline.next().map(|(idx, _)| idx).unwrap_or_default();
151                    for lint in lints {
152                        disabled_ranges.entry(lint).or_default().push(DisabledRange {
153                            start,
154                            end,
155                            loose: false,
156                        })
157                    }
158                }
159                InlineConfigItem::DisableNextLine(lints) => {
160                    let offset = sp.hi().to_usize();
161                    let mut char_indices =
162                        src[offset..].char_indices().skip_while(|(_, ch)| *ch != '\n').skip(1);
163                    if let Some((mut start, _)) = char_indices.next() {
164                        start += offset;
165                        let end = char_indices
166                            .find(|(_, ch)| *ch == '\n')
167                            .map(|(idx, _)| offset + idx + 1)
168                            .unwrap_or(src.len());
169                        for lint in lints {
170                            disabled_ranges.entry(lint).or_default().push(DisabledRange {
171                                start,
172                                end,
173                                loose: false,
174                            })
175                        }
176                    }
177                }
178                InlineConfigItem::DisableStart(lints) => {
179                    for lint in lints {
180                        disabled_blocks
181                            .entry(lint)
182                            .and_modify(|(_, depth)| *depth += 1)
183                            .or_insert((sp.hi().to_usize(), 1));
184                    }
185                }
186                InlineConfigItem::DisableEnd(lints) => {
187                    for lint in lints {
188                        if let Some((start, depth)) = disabled_blocks.get_mut(&lint) {
189                            *depth = depth.saturating_sub(1);
190
191                            if *depth == 0 {
192                                let start = *start;
193                                _ = disabled_blocks.remove(&lint);
194
195                                disabled_ranges.entry(lint).or_default().push(DisabledRange {
196                                    start,
197                                    end: sp.lo().to_usize(),
198                                    loose: false,
199                                })
200                            }
201                        }
202                    }
203                }
204            }
205        }
206
207        for (lint, (start, _)) in disabled_blocks {
208            disabled_ranges.entry(lint).or_default().push(DisabledRange {
209                start,
210                end: src.len(),
211                loose: false,
212            });
213        }
214
215        Self { disabled_ranges }
216    }
217
218    /// Check if the lint location is in a disabled range.
219    #[inline]
220    pub fn is_disabled(&self, span: Span, lint: &str) -> bool {
221        if let Some(ranges) = self.disabled_ranges.get(lint) {
222            return ranges.iter().any(|range| range.includes(span.to_range()));
223        }
224
225        if let Some(ranges) = self.disabled_ranges.get("all") {
226            return ranges.iter().any(|range| range.includes(span.to_range()));
227        }
228
229        false
230    }
231}

233/// An AST visitor that finds the first `Item` that starts after a given offset.
234#[derive(Debug, Default)]
235struct NextItemFinder<'ast> {
236    /// The offset to search after.
237    offset: usize,
238    _pd: PhantomData<&'ast ()>,
239}
240
241impl<'ast> NextItemFinder<'ast> {
242    fn new(offset: usize) -> Self {
243        Self { offset, _pd: PhantomData }
244    }
245
246    /// Finds the next AST item which a span that begins after the `offset`.
247    fn find(&mut self, ast: &'ast SourceUnit<'ast>) -> Option<Span> {
248        match self.visit_source_unit(ast) {
249            ControlFlow::Break(span) => Some(span),
250            ControlFlow::Continue(()) => None,
251        }
252    }
253}
254
255impl<'ast> Visit<'ast> for NextItemFinder<'ast> {
256    type BreakValue = Span;
257
258    fn visit_item(&mut self, item: &'ast Item<'ast>) -> ControlFlow<Self::BreakValue> {
259        // Check if this item starts after the offset.
260        if item.span.lo().to_usize() > self.offset {
261            return ControlFlow::Break(item.span);
262        }
263
264        // Otherwise, continue traversing inside this item.
265        self.walk_item(item)
266    }
267}
