16/// Trait representing a generic linter for analyzing and reporting issues in smart contract source
17/// code files. A linter can be implemented for any smart contract language supported by Foundry.
18///
19/// # Type Parameters
20///
21/// - `Language`: Represents the target programming language. Must implement the [`Language`] trait.
22/// - `Lint`: Represents the types of lints performed by the linter. Must implement the [`Lint`]
23///   trait.
24///
25/// # Required Methods
26///
27/// - `lint`: Scans the provided source files emitting a daignostic for lints found.
28pub trait Linter: Send + Sync + Clone {
29    type Language: Language;
30    type Lint: Lint;
31
32    fn lint(&self, input: &[PathBuf]);
33}
34
35pub trait Lint {
36    fn id(&self) -> &'static str;
37    fn severity(&self) -> Severity;
38    fn description(&self) -> &'static str;
39    fn help(&self) -> &'static str;
40}
41
42pub struct LintContext<'s> {
43    sess: &'s Session,
44    with_description: bool,
45    pub inline_config: InlineConfig,
46}
47
48impl<'s> LintContext<'s> {
49    pub fn new(sess: &'s Session, with_description: bool, config: InlineConfig) -> Self {
50        Self { sess, with_description, inline_config: config }
51    }
52
53    /// Helper method to emit diagnostics easily from passes
54    pub fn emit<L: Lint>(&self, lint: &'static L, span: Span) {
55        if self.inline_config.is_disabled(span, lint.id()) {
56            return;
57        }
58
59        let desc = if self.with_description { lint.description() } else { "" };
60        let diag: DiagBuilder<'_, ()> = self
61            .sess
62            .dcx
63            .diag(lint.severity().into(), desc)
64            .code(DiagId::new_str(lint.id()))
65            .span(MultiSpan::from_span(span))
66            .help(lint.help());
67
68        diag.emit();
69    }
70}

89    }

95    }

105/// Visitor struct for `EarlyLintPass`es
106pub struct EarlyLintVisitor<'a, 's, 'ast> {
107    pub ctx: &'a LintContext<'s>,
108    pub passes: &'a mut [Box<dyn EarlyLintPass<'ast> + 's>],
109}

111/// Extends the [`Visit`] trait functionality with a hook that can run after the initial traversal.
112impl<'s, 'ast> EarlyLintVisitor<'_, 's, 'ast>
113where
114    's: 'ast,
115{
116    pub fn post_source_unit(&mut self, ast: &'ast SourceUnit<'ast>) {
117        for pass in self.passes.iter_mut() {
118            pass.check_full_source_unit(self.ctx, ast);
119        }
120    }
121}
