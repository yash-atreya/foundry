26const METADATA_SUFFIX: &str = "metadata.json";
27const JSON_EXTENSION: &str = ".json";

29/// Possible mutation strategies to apply on a call sequence.
30#[derive(Debug, Clone)]
31enum MutationType {
32    /// Splice original call sequence.
33    Splice,
34    /// Repeat selected call several times.
35    Repeat,
36    /// Interleave calls from two random call sequences.
37    Interleave,
38    /// Replace prefix of the original call sequence with new calls.
39    Prefix,
40    /// Replace suffix of the original call sequence with new calls.
41    Suffix,
42    /// ABI mutate random args of selected call in sequence.
43    Abi,
44}
45
46/// Holds Corpus information.
47#[derive(Serialize)]
48struct Corpus {
49    // Unique corpus identifier.
50    uuid: Uuid,
51    // Total mutations of corpus as primary source.
52    total_mutations: usize,
53    // New coverage found as a result of mutating this corpus.
54    new_finds_produced: usize,
55    // Corpus call sequence.
56    #[serde(skip_serializing)]
57    tx_seq: Vec<BasicTxDetails>,
58}
59
60impl Corpus {
61    /// New corpus from given call sequence and corpus path to read uuid.
62    pub fn new(tx_seq: Vec<BasicTxDetails>, path: PathBuf) -> eyre::Result<Self> {
63        let uuid = if let Some(stem) = path.file_stem().and_then(|s| s.to_str()) {
64            Uuid::try_from(stem.strip_suffix(JSON_EXTENSION).unwrap_or(stem).to_string())?
65        } else {
66            Uuid::new_v4()
67        };
68        Ok(Self { uuid, total_mutations: 0, new_finds_produced: 0, tx_seq })
69    }
70
71    /// New corpus with given call sequence and new uuid.
72    pub fn from_tx_seq(tx_seq: Vec<BasicTxDetails>) -> Self {
73        Self { uuid: Uuid::new_v4(), total_mutations: 0, new_finds_produced: 0, tx_seq }
74    }
75}

77/// Invariant corpus manager.
78pub struct TxCorpusManager {
79    // Fuzzed calls generator.
80    tx_generator: BoxedStrategy<BasicTxDetails>,
81    // Call sequence mutation strategy type generator.
82    mutation_generator: BoxedStrategy<MutationType>,
83    // Path to invariant corpus directory. If None, sequences with new coverage are not persisted.
84    corpus_dir: Option<PathBuf>,
85    // Whether corpus to use gzip file compression and decompression.
86    corpus_gzip: bool,
87    // Number of mutations until entry marked as eligible to be flushed from in-memory corpus.
88    // Mutations will be perfored at least `corpus_min_mutations` times.
89    corpus_min_mutations: usize,
90    // Number of corpus that won't be evicted from memory.
91    corpus_min_size: usize,
92    // In-memory corpus, populated from persisted files and current runs.
93    // Mutation is performed on these.
94    in_memory_corpus: Vec<Corpus>,
95    // Identifier of current mutated entry.
96    current_mutated: Option<Uuid>,
97    // Number of failed replays from persisted corpus.
98    failed_replays: usize,
99}

101impl TxCorpusManager {
102    pub fn new(
103        invariant_config: &InvariantConfig,
104        test_name: &String,
105        fuzzed_contracts: &FuzzRunIdentifiedContracts,
106        tx_generator: BoxedStrategy<BasicTxDetails>,
107        executor: &Executor,
108        history_map: &mut [u8],
109    ) -> eyre::Result<Self> {
110        let mutation_generator = prop_oneof![
111            Just(MutationType::Splice),
112            Just(MutationType::Repeat),
113            Just(MutationType::Interleave),
114            Just(MutationType::Prefix),
115            Just(MutationType::Suffix),
116            Just(MutationType::Abi),
117        ]
118        .boxed();
119        let mut in_memory_corpus = vec![];
120        let corpus_gzip = invariant_config.corpus_gzip;
121        let corpus_min_mutations = invariant_config.corpus_min_mutations;
122        let corpus_min_size = invariant_config.corpus_min_size;
123        let mut failed_replays = 0;
124
125        // Early return if corpus dir / coverage guided fuzzing not configured.
126        let Some(corpus_dir) = &invariant_config.corpus_dir else {
127            return Ok(Self {
128                tx_generator,
129                mutation_generator,
130                corpus_dir: None,
131                corpus_gzip,
132                corpus_min_mutations,
133                corpus_min_size,
134                in_memory_corpus,
135                current_mutated: None,
136                failed_replays,
137            })
138        };
139
140        // Ensure corpus dir for invariant function is created.
141        let corpus_dir = corpus_dir.join(test_name);
142        if !corpus_dir.is_dir() {
143            foundry_common::fs::create_dir_all(&corpus_dir)?;
144        }
145
146        let fuzzed_contracts = fuzzed_contracts.targets.lock();
147
148        for entry in std::fs::read_dir(&corpus_dir)? {
149            let path = entry?.path();
150            if path.is_file() {
151                if let Some(name) = path.file_name().and_then(|s| s.to_str()) {
152                    // Ignore metadata files
153                    if name.contains(METADATA_SUFFIX) {
154                        continue
155                    }
156                }
157            }
158
159            let read_corpus_result = match path.extension().and_then(|ext| ext.to_str()) {
160                Some("gz") => foundry_common::fs::read_json_gzip_file::<Vec<BasicTxDetails>>(&path),
161                _ => foundry_common::fs::read_json_file::<Vec<BasicTxDetails>>(&path),
162            };
163
164            let Ok(tx_seq) = read_corpus_result else {
165                trace!(target: "corpus", "failed to load corpus from {}", path.display());
166                continue
167            };
168
169            if !tx_seq.is_empty() {
170                // Warm up history map from loaded sequences.
171                let mut executor = executor.clone();
172                for tx in &tx_seq {
173                    let mut call_result = executor
174                        .call_raw(
175                            tx.sender,
176                            tx.call_details.target,
177                            tx.call_details.calldata.clone(),
178                            U256::ZERO,
179                        )
180                        .map_err(|e| eyre!(format!("Could not make raw evm call: {e}")))?;
181
182                    if fuzzed_contracts.can_replay(tx) {
183                        call_result.merge_edge_coverage(history_map);
184                        executor.commit(&mut call_result);
185                    } else {
186                        failed_replays += 1;
187                    }
188                }
189
190                trace!(
191                    target: "corpus",
192                    "load sequence with len {} from corpus file {}",
193                    tx_seq.len(),
194                    path.display()
195                );
196
197                // Populate in memory corpus with sequence from corpus file.
198                in_memory_corpus.push(Corpus::new(tx_seq, path)?);
199            }
200        }
201
202        Ok(Self {
203            tx_generator,
204            mutation_generator,
205            corpus_dir: Some(corpus_dir),
206            corpus_gzip,
207            corpus_min_mutations,
208            corpus_min_size,
209            in_memory_corpus,
210            current_mutated: None,
211            failed_replays,
212        })
213    }

215    /// Collects inputs from given invariant run, if new coverage produced.
216    /// Persists call sequence (if corpus directory is configured) and updates in-memory corpus.
217    pub fn collect_inputs(&mut self, test_run: &InvariantTestRun) {
218        // Early return if corpus dir / coverage guided fuzzing is not configured.
219        let Some(corpus_dir) = &self.corpus_dir else {
220            return;
221        };
222
223        // Update stats of current mutated primary corpus.
224        if let Some(uuid) = &self.current_mutated {
225            if let Some(corpus) =
226                self.in_memory_corpus.iter_mut().find(|corpus| corpus.uuid.eq(uuid))
227            {
228                corpus.total_mutations += 1;
229                if test_run.new_coverage {
230                    corpus.new_finds_produced += 1
231                }
232
233                trace!(
234                    target: "corpus",
235                    "updated corpus {}, total mutations: {}, new finds: {}",
236                    corpus.uuid, corpus.total_mutations, corpus.new_finds_produced
237                );
238            }
239
240            self.current_mutated = None;
241        }
242
243        // Collect inputs only if current run produced new coverage.
244        if !test_run.new_coverage {
245            return;
246        }
247
248        let corpus = Corpus::from_tx_seq(test_run.inputs.clone());
249        let corpus_uuid = corpus.uuid;
250
251        // Persist to disk if corpus dir is configured.
252        let write_result = if self.corpus_gzip {
253            foundry_common::fs::write_json_gzip_file(
254                corpus_dir.join(format!("{corpus_uuid}{JSON_EXTENSION}.gz")).as_path(),
255                &corpus.tx_seq,
256            )
257        } else {
258            foundry_common::fs::write_json_file(
259                corpus_dir.join(format!("{corpus_uuid}{JSON_EXTENSION}")).as_path(),
260                &corpus.tx_seq,
261            )
262        };
263
264        if let Err(err) = write_result {
265            debug!(target: "corpus", %err, "Failed to record call sequence {:?}", &corpus.tx_seq);
266        } else {
267            trace!(
268                target: "corpus",
269                "persisted {} inputs for new coverage in {corpus_uuid} corpus",
270                &corpus.tx_seq.len()
271            );
272        }
273
274        // This includes reverting txs in the corpus and `can_continue` removes
275        // them. We want this as it is new coverage and may help reach the other branch.
276        self.in_memory_corpus.push(corpus);
277    }

279    /// Generates new call sequence from in memory corpus. Evicts oldest corpus mutated more than
280    /// configured max mutations value.
281    pub fn new_sequence(&mut self, test: &InvariantTest) -> eyre::Result<Vec<BasicTxDetails>> {
282        let mut new_seq = vec![];
283        let test_runner = &mut test.execution_data.borrow_mut().branch_runner;
284
285        // Early return with first_input only if corpus dir / coverage guided fuzzing not
286        // configured.
287        let Some(corpus_dir) = &self.corpus_dir else {
288            new_seq.push(self.new_tx(test_runner)?);
289            return Ok(new_seq);
290        };
291
292        if !self.in_memory_corpus.is_empty() {
293            // Flush oldest corpus mutated more than configured max mutations unless they are
294            // producing new finds more than 1/3 of the time.
295            let should_evict = self.in_memory_corpus.len() > self.corpus_min_size.max(1);
296            if should_evict {
297                if let Some(index) = self.in_memory_corpus.iter().position(|corpus| {
298                    corpus.total_mutations > self.corpus_min_mutations &&
299                        (corpus.new_finds_produced as f64 / corpus.total_mutations as f64) < 0.3
300                }) {
301                    let corpus = self.in_memory_corpus.get(index).unwrap();
302                    let uuid = corpus.uuid;
303                    debug!(target: "corpus", "evict corpus {uuid}");
304
305                    // Flush to disk the seed metadata at the time of eviction.
306                    let eviction_time = SystemTime::now()
307                        .duration_since(UNIX_EPOCH)
308                        .expect("Time went backwards")
309                        .as_secs();
310                    foundry_common::fs::write_json_file(
311                        corpus_dir
312                            .join(format!("{uuid}-{eviction_time}-{METADATA_SUFFIX}"))
313                            .as_path(),
314                        &corpus,
315                    )?;
316                    // Remove corpus from memory.
317                    self.in_memory_corpus.remove(index);
318                }
319            }
320
321            let mutation_type = self
322                .mutation_generator
323                .new_tree(test_runner)
324                .expect("Could not generate mutation type")
325                .current();
326            let rng = test_runner.rng();
327            let corpus_len = self.in_memory_corpus.len();
328            let primary = &self.in_memory_corpus[rng.random_range(0..corpus_len)];
329            let secondary = &self.in_memory_corpus[rng.random_range(0..corpus_len)];
330
331            match mutation_type {
332                MutationType::Splice => {
333                    trace!(target: "corpus", "splice {} and {}", primary.uuid, secondary.uuid);
334                    if should_evict {
335                        self.current_mutated = Some(primary.uuid);
336                    }
337                    let start1 = rng.random_range(0..primary.tx_seq.len());
338                    let end1 = rng.random_range(start1..primary.tx_seq.len());
339
340                    let start2 = rng.random_range(0..secondary.tx_seq.len());
341                    let end2 = rng.random_range(start2..secondary.tx_seq.len());
342
343                    for tx in primary.tx_seq.iter().take(end1).skip(start1) {
344                        new_seq.push(tx.clone());
345                    }
346                    for tx in secondary.tx_seq.iter().take(end2).skip(start2) {
347                        new_seq.push(tx.clone());
348                    }
349                }
350                MutationType::Repeat => {
351                    let corpus = if rng.random::<bool>() { primary } else { secondary };
352                    trace!(target: "corpus", "repeat {}", corpus.uuid);
353                    if should_evict {
354                        self.current_mutated = Some(corpus.uuid);
355                    }
356                    new_seq = corpus.tx_seq.clone();
357                    let start = rng.random_range(0..corpus.tx_seq.len());
358                    let end = rng.random_range(start..corpus.tx_seq.len());
359                    let item_idx = rng.random_range(0..corpus.tx_seq.len());
360                    let repeated = vec![new_seq[item_idx].clone(); end - start];
361                    new_seq.splice(start..end, repeated);
362                }
363                MutationType::Interleave => {
364                    trace!(target: "corpus", "interleave {} with {}", primary.uuid, secondary.uuid);
365                    if should_evict {
366                        self.current_mutated = Some(primary.uuid);
367                    }
368                    for (tx1, tx2) in primary.tx_seq.iter().zip(secondary.tx_seq.iter()) {
369                        // chunks?
370                        let tx = if rng.random::<bool>() { tx1.clone() } else { tx2.clone() };
371                        new_seq.push(tx);
372                    }
373                }
374                MutationType::Prefix => {
375                    let corpus = if rng.random::<bool>() { primary } else { secondary };
376                    trace!(target: "corpus", "overwrite prefix of {}", corpus.uuid);
377                    if should_evict {
378                        self.current_mutated = Some(corpus.uuid);
379                    }
380                    new_seq = corpus.tx_seq.clone();
381                    for i in 0..rng.random_range(0..=new_seq.len()) {
382                        new_seq[i] = self.new_tx(test_runner)?;
383                    }
384                }
385                MutationType::Suffix => {
386                    let corpus = if rng.random::<bool>() { primary } else { secondary };
387                    trace!(target: "corpus", "overwrite suffix of {}", corpus.uuid);
388                    if should_evict {
389                        self.current_mutated = Some(corpus.uuid);
390                    }
391                    new_seq = corpus.tx_seq.clone();
392                    for i in new_seq.len() - rng.random_range(0..new_seq.len())..corpus.tx_seq.len()
393                    {
394                        new_seq[i] = self.new_tx(test_runner)?;
395                    }
396                }
397                MutationType::Abi => {
398                    let targets = test.targeted_contracts.targets.lock();
399                    let corpus = if rng.random::<bool>() { primary } else { secondary };
400                    trace!(target: "corpus", "ABI mutate args of {}", corpus.uuid);
401                    if should_evict {
402                        self.current_mutated = Some(corpus.uuid);
403                    }
404                    new_seq = corpus.tx_seq.clone();
405
406                    let idx = rng.random_range(0..new_seq.len());
407                    let tx = new_seq.get_mut(idx).unwrap();
408                    if let (_, Some(function)) = targets.fuzzed_artifacts(tx) {
409                        // TODO add call_value to call details and mutate it as well as sender some
410                        // of the time
411                        if !function.inputs.is_empty() {
412                            let mut new_function = function.clone();
413                            let mut arg_mutation_rounds =
414                                rng.random_range(0..=function.inputs.len()).max(1);
415                            let round_arg_idx: Vec<usize> = if function.inputs.len() <= 1 {
416                                vec![0]
417                            } else {
418                                (0..arg_mutation_rounds)
419                                    .map(|_| {
420                                        test_runner.rng().random_range(0..function.inputs.len())
421                                    })
422                                    .collect()
423                            };
424                            // TODO mutation strategy for individual ABI types
425                            let mut prev_inputs = function
426                                .abi_decode_input(&tx.call_details.calldata[4..])
427                                .expect("fuzzed_artifacts returned wrong sig");
428                            // For now, only new inputs are generated, no existing inputs are
429                            // mutated.
430                            let mut gen_input = |input: &alloy_json_abi::Param| {
431                                fuzz_param_from_state(
432                                    &input.selector_type().parse().unwrap(),
433                                    &test.fuzz_state,
434                                )
435                                .new_tree(test_runner)
436                                .expect("Could not generate case")
437                                .current()
438                            };
439
440                            while arg_mutation_rounds > 0 {
441                                let idx = round_arg_idx[arg_mutation_rounds - 1];
442                                let input = new_function
443                                    .inputs
444                                    .get_mut(idx)
445                                    .expect("Could not get input to mutate");
446                                let new_input = gen_input(input);
447                                prev_inputs[idx] = new_input;
448                                arg_mutation_rounds -= 1;
449                            }
450
451                            tx.call_details.calldata = new_function
452                                .abi_encode_input(&prev_inputs)
453                                .map_err(|e| eyre!(e.to_string()))?
454                                .into();
455                        }
456                    }
457                }
458            }
459        }
460
461        // Make sure sequence contains at least one tx to start fuzzing from.
462        if new_seq.is_empty() {
463            new_seq.push(self.new_tx(test_runner)?);
464        }
465        trace!(target: "corpus", "new sequence of {} calls generated", new_seq.len());
466
467        Ok(new_seq)
468    }

470    /// Returns the next call to be used in call sequence.
471    /// If coverage guided fuzzing is not configured or if previous input was discarded then this is
472    /// a new tx from strategy.
473    /// If running with coverage guided fuzzing it returns a new call only when sequence
474    /// does not have enough entries, or randomly. Otherwise, returns the next call from initial
475    /// sequence.
476    pub fn generate_next_input(
477        &mut self,
478        test: &InvariantTest,
479        sequence: &[BasicTxDetails],
480        discarded: bool,
481        depth: usize,
482    ) -> eyre::Result<BasicTxDetails> {
483        let test_runner = &mut test.execution_data.borrow_mut().branch_runner;
484
485        // Early return with new input if corpus dir / coverage guided fuzzing not configured or if
486        // call was discarded.
487        if self.corpus_dir.is_none() || discarded {
488            return self.new_tx(test_runner)
489        }
490
491        // When running with coverage guided fuzzing enabled then generate new sequence if initial
492        // sequence's length is less than depth or randomly, to occasionally intermix new txs.
493        if depth > sequence.len().saturating_sub(1) || test_runner.rng().random_ratio(1, 10) {
494            return self.new_tx(test_runner)
495        }
496
497        // Continue with the next call initial sequence
498        Ok(sequence[depth].clone())
499    }

501    /// Generates single call from invariant strategy.
502    pub fn new_tx(&mut self, test_runner: &mut TestRunner) -> eyre::Result<BasicTxDetails> {
503        Ok(self
504            .tx_generator
505            .new_tree(test_runner)
506            .map_err(|_| eyre!("Could not generate case"))?
507            .current())
508    }
