147    pub fuzz_state: EvmFuzzState,

154impl InvariantTest {
155    /// Instantiates an invariant test.
156    pub fn new(
157        fuzz_state: EvmFuzzState,
158        targeted_contracts: FuzzRunIdentifiedContracts,
159        failures: InvariantFailures,
160        last_call_results: Option<RawCallResult>,
161        branch_runner: TestRunner,
162    ) -> Self {
163        let mut fuzz_cases = vec![];
164        if last_call_results.is_none() {
165            fuzz_cases.push(FuzzedCases::new(vec![]));
166        }
167        let execution_data = RefCell::new(InvariantTestData {
168            fuzz_cases,
169            failures,
170            last_run_inputs: vec![],
171            gas_report_traces: vec![],
172            last_call_results,
173            line_coverage: None,
174            metrics: Map::default(),
175            branch_runner,
176        });
177        Self { fuzz_state, targeted_contracts, execution_data }
178    }
179
180    /// Returns number of invariant test reverts.
181    pub fn reverts(&self) -> usize {
182        self.execution_data.borrow().failures.reverts
183    }
184
185    /// Whether invariant test has errors or not.
186    pub fn has_errors(&self) -> bool {
187        self.execution_data.borrow().failures.error.is_some()
188    }
189
190    /// Set invariant test error.
191    pub fn set_error(&self, error: InvariantFuzzError) {
192        self.execution_data.borrow_mut().failures.error = Some(error);
193    }
194
195    /// Set last invariant test call results.
196    pub fn set_last_call_results(&self, call_result: Option<RawCallResult>) {
197        self.execution_data.borrow_mut().last_call_results = call_result;
198    }
199
200    /// Set last invariant run call sequence.
201    pub fn set_last_run_inputs(&self, inputs: &Vec<BasicTxDetails>) {
202        self.execution_data.borrow_mut().last_run_inputs.clone_from(inputs);
203    }
204
205    /// Merge current collected line coverage with the new coverage from last fuzzed call.
206    pub fn merge_coverage(&self, new_coverage: Option<HitMaps>) {
207        HitMaps::merge_opt(&mut self.execution_data.borrow_mut().line_coverage, new_coverage);
208    }
209
210    /// Update metrics for a fuzzed selector, extracted from tx details.
211    /// Always increments number of calls; discarded runs (through assume cheatcodes) are tracked
212    /// separated from reverts.
213    pub fn record_metrics(&self, tx_details: &BasicTxDetails, reverted: bool, discarded: bool) {
214        if let Some(metric_key) =
215            self.targeted_contracts.targets.lock().fuzzed_metric_key(tx_details)
216        {
217            let test_metrics = &mut self.execution_data.borrow_mut().metrics;
218            let invariant_metrics = test_metrics.entry(metric_key).or_default();
219            invariant_metrics.calls += 1;
220            if discarded {
221                invariant_metrics.discards += 1;
222            } else if reverted {
223                invariant_metrics.reverts += 1;
224            }
225        }
226    }
227
228    /// End invariant test run by collecting results, cleaning collected artifacts and reverting
229    /// created fuzz state.
230    pub fn end_run(&self, run: InvariantTestRun, gas_samples: usize) {
231        // We clear all the targeted contracts created during this run.
232        self.targeted_contracts.clear_created_contracts(run.created_contracts);
233
234        let mut invariant_data = self.execution_data.borrow_mut();
235        if invariant_data.gas_report_traces.len() < gas_samples {
236            invariant_data
237                .gas_report_traces
238                .push(run.run_traces.into_iter().map(|arena| arena.arena).collect());
239        }
240        invariant_data.fuzz_cases.push(FuzzedCases::new(run.fuzz_runs));
241
242        // Revert state to not persist values between runs.
243        self.fuzz_state.revert();
244    }

283/// Wrapper around any [`Executor`] implementer which provides fuzzing support using [`proptest`].
284///
285/// After instantiation, calling `invariant_fuzz` will proceed to hammer the deployed smart
286/// contracts with inputs, until it finds a counterexample sequence. The provided [`TestRunner`]
287/// contains all the configuration which can be overridden via [environment
288/// variables](proptest::test_runner::Config)
289pub struct InvariantExecutor<'a> {
290    pub executor: Executor,
291    /// Proptest runner.
292    runner: TestRunner,
293    /// The invariant configuration
294    config: InvariantConfig,
295    /// Contracts deployed with `setUp()`
296    setup_contracts: &'a ContractsByAddress,
297    /// Contracts that are part of the project but have not been deployed yet. We need the bytecode
298    /// to identify them from the stateset changes.
299    project_contracts: &'a ContractsByArtifact,
300    /// Filters contracts to be fuzzed through their artifact identifiers.
301    artifact_filters: ArtifactFilters,
302    /// History of binned hitcount of edges seen during fuzzing.
303    history_map: Vec<u8>,
304}
305const COVERAGE_MAP_SIZE: usize = 65536;
306
307impl<'a> InvariantExecutor<'a> {
308    /// Instantiates a fuzzed executor EVM given a testrunner
309    pub fn new(
310        executor: Executor,
311        runner: TestRunner,
312        config: InvariantConfig,
313        setup_contracts: &'a ContractsByAddress,
314        project_contracts: &'a ContractsByArtifact,
315    ) -> Self {
316        Self {
317            executor,
318            runner,
319            config,
320            setup_contracts,
321            project_contracts,
322            artifact_filters: ArtifactFilters::default(),
323            history_map: vec![0u8; COVERAGE_MAP_SIZE],
324        }
325    }

327    /// Fuzzes any deployed contract and checks any broken invariant at `invariant_address`.
328    pub fn invariant_fuzz(
329        &mut self,
330        invariant_contract: InvariantContract<'_>,
331        fuzz_fixtures: &FuzzFixtures,
332        deployed_libs: &[Address],
333        progress: Option<&ProgressBar>,
334    ) -> Result<InvariantFuzzTestResult> {
335        // Throw an error to abort test run if the invariant function accepts input params
336        if !invariant_contract.invariant_function.inputs.is_empty() {
337            return Err(eyre!("Invariant test function should have no inputs"));
338        }
339
340        let (invariant_test, mut corpus_manager) =
341            self.prepare_test(&invariant_contract, fuzz_fixtures, deployed_libs)?;
342
343        // Start timer for this invariant test.
344        let mut runs = 0;
345        let timer = FuzzTestTimer::new(self.config.timeout);
346        let continue_campaign = |runs: u32| {
347            // If timeout is configured, then perform invariant runs until expires.
348            if self.config.timeout.is_some() {
349                return !timer.is_timed_out()
350            }
351            // If no timeout configured then loop until configured runs.
352            runs < self.config.runs
353        };
354
355        'stop: while continue_campaign(runs) {
356            let initial_seq = corpus_manager.new_sequence(&invariant_test)?;
357
358            // Create current invariant run data.
359            let mut current_run = InvariantTestRun::new(
360                initial_seq[0].clone(),
361                // Before each run, we must reset the backend state.
362                self.executor.clone(),
363                self.config.depth as usize,
364            );
365
366            // We stop the run immediately if we have reverted, and `fail_on_revert` is set.
367            if self.config.fail_on_revert && invariant_test.reverts() > 0 {
368                return Err(eyre!("call reverted"))
369            }
370
371            while current_run.depth < self.config.depth {
372                // Check if the timeout has been reached.
373                if timer.is_timed_out() {
374                    // Since we never record a revert here the test is still considered
375                    // successful even though it timed out. We *want*
376                    // this behavior for now, so that's ok, but
377                    // future developers should be aware of this.
378                    break 'stop;
379                }
380
381                let tx = current_run
382                    .inputs
383                    .last()
384                    .ok_or_else(|| eyre!("no input generated to call fuzzed target."))?;
385
386                // Execute call from the randomly generated sequence without committing state.
387                // State is committed only if call is not a magic assume.
388                let mut call_result = current_run
389                    .executor
390                    .call_raw(
391                        tx.sender,
392                        tx.call_details.target,
393                        tx.call_details.calldata.clone(),
394                        U256::ZERO,
395                    )
396                    .map_err(|e| eyre!(format!("Could not make raw evm call: {e}")))?;
397
398                let discarded = call_result.result.as_ref() == MAGIC_ASSUME;
399                if self.config.show_metrics {
400                    invariant_test.record_metrics(tx, call_result.reverted, discarded);
401                }
402
403                // Collect line coverage from last fuzzed call.
404                invariant_test.merge_coverage(call_result.line_coverage.clone());
405                // If coverage guided fuzzing is enabled then merge edge count with current history
406                // map and set new coverage in current run.
407                if self.config.corpus_dir.is_some() &&
408                    call_result.merge_edge_coverage(&mut self.history_map)
409                {
410                    current_run.new_coverage = true;
411                }
412
413                if discarded {
414                    current_run.inputs.pop();
415                    current_run.assume_rejects_counter += 1;
416                    if current_run.assume_rejects_counter > self.config.max_assume_rejects {
417                        invariant_test.set_error(InvariantFuzzError::MaxAssumeRejects(
418                            self.config.max_assume_rejects,
419                        ));
420                        break 'stop;
421                    }
422                } else {
423                    // Commit executed call result.
424                    current_run.executor.commit(&mut call_result);
425
426                    // Collect data for fuzzing from the state changeset.
427                    // This step updates the state dictionary and therefore invalidates the
428                    // ValueTree in use by the current run. This manifestsitself in proptest
429                    // observing a different input case than what it was called with, and creates
430                    // inconsistencies whenever proptest tries to use the input case after test
431                    // execution.
432                    // See <https://github.com/foundry-rs/foundry/issues/9764>.
433                    let mut state_changeset = call_result.state_changeset.clone();
434                    if !call_result.reverted {
435                        collect_data(
436                            &invariant_test,
437                            &mut state_changeset,
438                            tx,
439                            &call_result,
440                            self.config.depth,
441                        );
442                    }
443
444                    // Collect created contracts and add to fuzz targets only if targeted contracts
445                    // are updatable.
446                    if let Err(error) =
447                        &invariant_test.targeted_contracts.collect_created_contracts(
448                            &state_changeset,
449                            self.project_contracts,
450                            self.setup_contracts,
451                            &self.artifact_filters,
452                            &mut current_run.created_contracts,
453                        )
454                    {
455                        warn!(target: "forge::test", "{error}");
456                    }
457                    current_run.fuzz_runs.push(FuzzCase {
458                        calldata: tx.call_details.calldata.clone(),
459                        gas: call_result.gas_used,
460                        stipend: call_result.stipend,
461                    });
462
463                    // Determine if test can continue or should exit.
464                    let result = can_continue(
465                        &invariant_contract,
466                        &invariant_test,
467                        &mut current_run,
468                        &self.config,
469                        call_result,
470                        &state_changeset,
471                    )
472                    .map_err(|e| eyre!(e.to_string()))?;
473                    if !result.can_continue || current_run.depth == self.config.depth - 1 {
474                        invariant_test.set_last_run_inputs(&current_run.inputs);
475                    }
476                    // If test cannot continue then stop current run and exit test suite.
477                    if !result.can_continue {
478                        break 'stop;
479                    }
480
481                    invariant_test.set_last_call_results(result.call_result);
482                    current_run.depth += 1;
483                }
484
485                current_run.inputs.push(corpus_manager.generate_next_input(
486                    &invariant_test,
487                    &initial_seq,
488                    discarded,
489                    current_run.depth as usize,
490                )?);
491            }
492
493            // Extend corpus with current run data.
494            corpus_manager.collect_inputs(&current_run);
495
496            // Call `afterInvariant` only if it is declared and test didn't fail already.
497            if invariant_contract.call_after_invariant && !invariant_test.has_errors() {
498                assert_after_invariant(
499                    &invariant_contract,
500                    &invariant_test,
501                    &current_run,
502                    &self.config,
503                )
504                .map_err(|_| eyre!("Failed to call afterInvariant"))?;
505            }
506
507            // End current invariant test run.
508            invariant_test.end_run(current_run, self.config.gas_report_samples as usize);
509
510            // If running with progress then increment completed runs.
511            if let Some(progress) = progress {
512                progress.inc(1);
513            }
514
515            runs += 1;
516        }
517
518        trace!(?fuzz_fixtures);
519        invariant_test.fuzz_state.log_stats();
520
521        let result = invariant_test.execution_data.into_inner();
522        Ok(InvariantFuzzTestResult {
523            error: result.failures.error,
524            cases: result.fuzz_cases,
525            reverts: result.failures.reverts,
526            last_run_inputs: result.last_run_inputs,
527            gas_report_traces: result.gas_report_traces,
528            line_coverage: result.line_coverage,
529            metrics: result.metrics,
530            failed_corpus_replays: corpus_manager.failed_replays(),
531        })
532    }

534    /// Prepares certain structures to execute the invariant tests:
535    /// * Invariant Fuzz Test.
536    /// * Invariant Corpus Manager.
537    fn prepare_test(
538        &mut self,
539        invariant_contract: &InvariantContract<'_>,
540        fuzz_fixtures: &FuzzFixtures,
541        deployed_libs: &[Address],
542    ) -> Result<(InvariantTest, TxCorpusManager)> {
543        // Finds out the chosen deployed contracts and/or senders.
544        self.select_contract_artifacts(invariant_contract.address)?;
545        let (targeted_senders, targeted_contracts) =
546            self.select_contracts_and_senders(invariant_contract.address)?;
547
548        // Stores fuzz state for use with [fuzz_calldata_from_state].
549        let fuzz_state = EvmFuzzState::new(
550            self.executor.backend().mem_db(),
551            self.config.dictionary,
552            deployed_libs,
553        );
554
555        // Creates the invariant strategy.
556        let strategy = invariant_strat(
557            fuzz_state.clone(),
558            targeted_senders,
559            targeted_contracts.clone(),
560            self.config.dictionary.dictionary_weight,
561            fuzz_fixtures.clone(),
562        )
563        .no_shrink();
564
565        // Allows `override_call_strat` to use the address given by the Fuzzer inspector during
566        // EVM execution.
567        let mut call_generator = None;
568        if self.config.call_override {
569            let target_contract_ref = Arc::new(RwLock::new(Address::ZERO));
570
571            call_generator = Some(RandomCallGenerator::new(
572                invariant_contract.address,
573                self.runner.clone(),
574                override_call_strat(
575                    fuzz_state.clone(),
576                    targeted_contracts.clone(),
577                    target_contract_ref.clone(),
578                    fuzz_fixtures.clone(),
579                ),
580                target_contract_ref,
581            ));
582        }
583
584        self.executor.inspector_mut().fuzzer =
585            Some(Fuzzer { call_generator, fuzz_state: fuzz_state.clone(), collect: true });
586
587        // Let's make sure the invariant is sound before actually starting the run:
588        // We'll assert the invariant in its initial state, and if it fails, we'll
589        // already know if we can early exit the invariant run.
590        // This does not count as a fuzz run. It will just register the revert.
591        let mut failures = InvariantFailures::new();
592        let last_call_results = assert_invariants(
593            invariant_contract,
594            &self.config,
595            &targeted_contracts,
596            &self.executor,
597            &[],
598            &mut failures,
599        )?;
600        if let Some(error) = failures.error {
601            return Err(eyre!(error.revert_reason().unwrap_or_default()))
602        }
603
604        let corpus_manager = TxCorpusManager::new(
605            &self.config,
606            &invariant_contract.invariant_function.name,
607            &targeted_contracts,
608            strategy.boxed(),
609            &self.executor,
610            &mut self.history_map,
611        )?;
612
613        let invariant_test = InvariantTest::new(
614            fuzz_state,
615            targeted_contracts,
616            failures,
617            last_call_results,
618            self.runner.clone(),
619        );
620
621        Ok((invariant_test, corpus_manager))
622    }

624    /// Fills the `InvariantExecutor` with the artifact identifier filters (in `path:name` string
625    /// format). They will be used to filter contracts after the `setUp`, and more importantly,
626    /// during the runs.
627    ///
628    /// Also excludes any contract without any mutable functions.
629    ///
630    /// Priority:
631    ///
632    /// targetArtifactSelectors > excludeArtifacts > targetArtifacts
633    pub fn select_contract_artifacts(&mut self, invariant_address: Address) -> Result<()> {
634        let targeted_artifact_selectors = self
635            .executor
636            .call_sol_default(invariant_address, &IInvariantTest::targetArtifactSelectorsCall {});
637
638        // Insert them into the executor `targeted_abi`.
639        for IInvariantTest::FuzzArtifactSelector { artifact, selectors } in
640            targeted_artifact_selectors
641        {
642            let identifier = self.validate_selected_contract(artifact, &selectors)?;
643            self.artifact_filters.targeted.entry(identifier).or_default().extend(selectors);
644        }
645
646        let targeted_artifacts = self
647            .executor
648            .call_sol_default(invariant_address, &IInvariantTest::targetArtifactsCall {});
649        let excluded_artifacts = self
650            .executor
651            .call_sol_default(invariant_address, &IInvariantTest::excludeArtifactsCall {});
652
653        // Insert `excludeArtifacts` into the executor `excluded_abi`.
654        for contract in excluded_artifacts {
655            let identifier = self.validate_selected_contract(contract, &[])?;
656
657            if !self.artifact_filters.excluded.contains(&identifier) {
658                self.artifact_filters.excluded.push(identifier);
659            }
660        }
661
662        // Exclude any artifact without mutable functions.
663        for (artifact, contract) in self.project_contracts.iter() {
664            if contract
665                .abi
666                .functions()
667                .filter(|func| {
668                    !matches!(
669                        func.state_mutability,
670                        alloy_json_abi::StateMutability::Pure |
671                            alloy_json_abi::StateMutability::View
672                    )
673                })
674                .count() ==
675                0 &&
676                !self.artifact_filters.excluded.contains(&artifact.identifier())
677            {
678                self.artifact_filters.excluded.push(artifact.identifier());
679            }
680        }
681
682        // Insert `targetArtifacts` into the executor `targeted_abi`, if they have not been seen
683        // before.
684        for contract in targeted_artifacts {
685            let identifier = self.validate_selected_contract(contract, &[])?;
686
687            if !self.artifact_filters.targeted.contains_key(&identifier) &&
688                !self.artifact_filters.excluded.contains(&identifier)
689            {
690                self.artifact_filters.targeted.insert(identifier, vec![]);
691            }
692        }
693        Ok(())
694    }

696    /// Makes sure that the contract exists in the project. If so, it returns its artifact
697    /// identifier.
698    fn validate_selected_contract(
699        &mut self,
700        contract: String,
701        selectors: &[FixedBytes<4>],
702    ) -> Result<String> {
703        if let Some((artifact, contract_data)) =
704            self.project_contracts.find_by_name_or_identifier(&contract)?
705        {
706            // Check that the selectors really exist for this contract.
707            for selector in selectors {
708                contract_data
709                    .abi
710                    .functions()
711                    .find(|func| func.selector().as_slice() == selector.as_slice())
712                    .wrap_err(format!("{contract} does not have the selector {selector:?}"))?;
713            }
714
715            return Ok(artifact.identifier())
716        }
717        eyre::bail!("{contract} not found in the project. Allowed format: `contract_name` or `contract_path:contract_name`.");
718    }

720    /// Selects senders and contracts based on the contract methods `targetSenders() -> address[]`,
721    /// `targetContracts() -> address[]` and `excludeContracts() -> address[]`.
722    pub fn select_contracts_and_senders(
723        &self,
724        to: Address,
725    ) -> Result<(SenderFilters, FuzzRunIdentifiedContracts)> {
726        let targeted_senders =
727            self.executor.call_sol_default(to, &IInvariantTest::targetSendersCall {});
728        let mut excluded_senders =
729            self.executor.call_sol_default(to, &IInvariantTest::excludeSendersCall {});
730        // Extend with default excluded addresses - https://github.com/foundry-rs/foundry/issues/4163
731        excluded_senders.extend([
732            CHEATCODE_ADDRESS,
733            HARDHAT_CONSOLE_ADDRESS,
734            DEFAULT_CREATE2_DEPLOYER,
735        ]);
736        // Extend with precompiles - https://github.com/foundry-rs/foundry/issues/4287
737        excluded_senders.extend(PRECOMPILES);
738        let sender_filters = SenderFilters::new(targeted_senders, excluded_senders);
739
740        let selected = self.executor.call_sol_default(to, &IInvariantTest::targetContractsCall {});
741        let excluded = self.executor.call_sol_default(to, &IInvariantTest::excludeContractsCall {});
742
743        let contracts = self
744            .setup_contracts
745            .iter()
746            .filter(|&(addr, (identifier, _))| {
747                // Include to address if explicitly set as target.
748                if *addr == to && selected.contains(&to) {
749                    return true;
750                }
751
752                *addr != to &&
753                    *addr != CHEATCODE_ADDRESS &&
754                    *addr != HARDHAT_CONSOLE_ADDRESS &&
755                    (selected.is_empty() || selected.contains(addr)) &&
756                    (excluded.is_empty() || !excluded.contains(addr)) &&
757                    self.artifact_filters.matches(identifier)
758            })
759            .map(|(addr, (identifier, abi))| {
760                (*addr, TargetedContract::new(identifier.clone(), abi.clone()))
761            })
762            .collect();
763        let mut contracts = TargetedContracts { inner: contracts };
764
765        self.target_interfaces(to, &mut contracts)?;
766
767        self.select_selectors(to, &mut contracts)?;
768
769        // There should be at least one contract identified as target for fuzz runs.
770        if contracts.is_empty() {
771            eyre::bail!("No contracts to fuzz.");
772        }
773
774        Ok((sender_filters, FuzzRunIdentifiedContracts::new(contracts, selected.is_empty())))
775    }

777    /// Extends the contracts and selectors to fuzz with the addresses and ABIs specified in
778    /// `targetInterfaces() -> (address, string[])[]`. Enables targeting of addresses that are
779    /// not deployed during `setUp` such as when fuzzing in a forked environment. Also enables
780    /// targeting of delegate proxies and contracts deployed with `create` or `create2`.
781    pub fn target_interfaces(
782        &self,
783        invariant_address: Address,
784        targeted_contracts: &mut TargetedContracts,
785    ) -> Result<()> {
786        let interfaces = self
787            .executor
788            .call_sol_default(invariant_address, &IInvariantTest::targetInterfacesCall {});
789
790        // Since `targetInterfaces` returns a tuple array there is no guarantee
791        // that the addresses are unique this map is used to merge functions of
792        // the specified interfaces for the same address. For example:
793        // `[(addr1, ["IERC20", "IOwnable"])]` and `[(addr1, ["IERC20"]), (addr1, ("IOwnable"))]`
794        // should be equivalent.
795        let mut combined = TargetedContracts::new();
796
797        // Loop through each address and its associated artifact identifiers.
798        // We're borrowing here to avoid taking full ownership.
799        for IInvariantTest::FuzzInterface { addr, artifacts } in &interfaces {
800            // Identifiers are specified as an array, so we loop through them.
801            for identifier in artifacts {
802                // Try to find the contract by name or identifier in the project's contracts.
803                if let Some(abi) = self.project_contracts.find_abi_by_name_or_identifier(identifier)
804                {
805                    combined
806                        // Check if there's an entry for the given key in the 'combined' map.
807                        .entry(*addr)
808                        // If the entry exists, extends its ABI with the function list.
809                        .and_modify(|entry| {
810                            // Extend the ABI's function list with the new functions.
811                            entry.abi.functions.extend(abi.functions.clone());
812                        })
813                        // Otherwise insert it into the map.
814                        .or_insert_with(|| TargetedContract::new(identifier.to_string(), abi));
815                }
816            }
817        }
818
819        targeted_contracts.extend(combined.inner);
820
821        Ok(())
822    }

824    /// Selects the functions to fuzz based on the contract method `targetSelectors()` and
825    /// `targetArtifactSelectors()`.
826    pub fn select_selectors(
827        &self,
828        address: Address,
829        targeted_contracts: &mut TargetedContracts,
830    ) -> Result<()> {
831        if let Some(target) = targeted_contracts.get(&address) {
832            // If test contract is a target, then include only state-changing functions
833            // that are not reserved.
834            let selectors: Vec<_> = target
835                .abi
836                .functions()
837                .filter_map(|func| {
838                    if matches!(
839                        func.state_mutability,
840                        alloy_json_abi::StateMutability::Pure |
841                            alloy_json_abi::StateMutability::View
842                    ) || func.is_reserved()
843                    {
844                        None
845                    } else {
846                        Some(func.selector())
847                    }
848                })
849                .collect();
850            self.add_address_with_functions(address, &selectors, false, targeted_contracts)?;
851        }
852
853        for (address, (identifier, _)) in self.setup_contracts {
854            if let Some(selectors) = self.artifact_filters.targeted.get(identifier) {
855                self.add_address_with_functions(*address, selectors, false, targeted_contracts)?;
856            }
857        }
858
859        // Collect contract functions marked as target for fuzzing campaign.
860        let selectors =
861            self.executor.call_sol_default(address, &IInvariantTest::targetSelectorsCall {});
862        for IInvariantTest::FuzzSelector { addr, selectors } in selectors {
863            self.add_address_with_functions(addr, &selectors, false, targeted_contracts)?;
864        }
865
866        // Collect contract functions excluded from fuzzing campaign.
867        let excluded_selectors =
868            self.executor.call_sol_default(address, &IInvariantTest::excludeSelectorsCall {});
869        for IInvariantTest::FuzzSelector { addr, selectors } in excluded_selectors {
870            self.add_address_with_functions(addr, &selectors, true, targeted_contracts)?;
871        }
872
873        Ok(())
874    }
875
876    /// Adds the address and fuzzed or excluded functions to `TargetedContracts`.
877    fn add_address_with_functions(
878        &self,
879        address: Address,
880        selectors: &[Selector],
881        should_exclude: bool,
882        targeted_contracts: &mut TargetedContracts,
883    ) -> eyre::Result<()> {
884        // Do not add address in target contracts if no function selected.
885        if selectors.is_empty() {
886            return Ok(())
887        }
888
889        let contract = match targeted_contracts.entry(address) {
890            Entry::Occupied(entry) => entry.into_mut(),
891            Entry::Vacant(entry) => {
892                let (identifier, abi) = self.setup_contracts.get(&address).ok_or_else(|| {
893                    eyre::eyre!(
894                        "[{}] address does not have an associated contract: {}",
895                        if should_exclude { "excludeSelectors" } else { "targetSelectors" },
896                        address
897                    )
898                })?;
899                entry.insert(TargetedContract::new(identifier.clone(), abi.clone()))
900            }
901        };
902        contract.add_selectors(selectors.iter().copied(), should_exclude)?;
903        Ok(())
904    }
905}

907/// Collects data from call for fuzzing. However, it first verifies that the sender is not an EOA
908/// before inserting it into the dictionary. Otherwise, we flood the dictionary with
909/// randomly generated addresses.
910fn collect_data(
911    invariant_test: &InvariantTest,
912    state_changeset: &mut HashMap<Address, Account>,
913    tx: &BasicTxDetails,
914    call_result: &RawCallResult,
915    run_depth: u32,
916) {
917    // Verify it has no code.
918    let mut has_code = false;
919    if let Some(Some(code)) =
920        state_changeset.get(&tx.sender).map(|account| account.info.code.as_ref())
921    {
922        has_code = !code.is_empty();
923    }
924
925    // We keep the nonce changes to apply later.
926    let mut sender_changeset = None;
927    if !has_code {
928        sender_changeset = state_changeset.remove(&tx.sender);
929    }
930
931    // Collect values from fuzzed call result and add them to fuzz dictionary.
932    invariant_test.fuzz_state.collect_values_from_call(
933        &invariant_test.targeted_contracts,
934        tx,
935        &call_result.result,
936        &call_result.logs,
937        &*state_changeset,
938        run_depth,
939    );
940
941    // Re-add changes
942    if let Some(changed) = sender_changeset {
943        state_changeset.insert(tx.sender, changed);
944    }
945}

947/// Calls the `afterInvariant()` function on a contract.
948/// Returns call result and if call succeeded.
949/// The state after the call is not persisted.
950pub(crate) fn call_after_invariant_function(
951    executor: &Executor,
952    to: Address,
953) -> Result<(RawCallResult, bool), EvmError> {
954    let calldata = Bytes::from_static(&IInvariantTest::afterInvariantCall::SELECTOR);
955    let mut call_result = executor.call_raw(CALLER, to, calldata, U256::ZERO)?;
956    let success = executor.is_raw_call_mut_success(to, &mut call_result, false);
957    Ok((call_result, success))
958}
